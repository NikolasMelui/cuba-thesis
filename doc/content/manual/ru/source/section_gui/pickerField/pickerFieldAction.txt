protected class SomeAction extends AbstractAction implements ShortcutAction{

    protected SomeAction(String id) {
        super(id);
    }

    @Override
    public KeyCombination getKeyCombination() {
        return new KeyCombination(Key.ARROW_UP,Modifier.CTRL);
    }

    @Override
    public void actionPerform(Component component) {
        pickerField4.setValue("UP");
    }

    @Override
    public String getCaption() {
        return "";
    }

    @Override
    public String getIcon() {
        return "icons/up.png";
    }
}
