public class CheckBoxSample extends AbstractWindow {
    @Inject
    private CheckBox chBox;

    @Override
    public void init(Map<String, Object> params) {
        chBox.addListener(new ValueListener<Object>() {
            @Override
            public void valueChanged(Object source, String property, Object prevValue, Object value) {
                if (chBox.getValue().equals(true)) {
                    showNotification(getMessage("notificationMessage"), NotificationType.HUMANIZED);
                } else {
                    showNotification(getMessage("notificationMessage2"), NotificationType.HUMANIZED);
                }
            }
        });
    }
}
